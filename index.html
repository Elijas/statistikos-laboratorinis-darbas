random.uniform = function(n, seed=12, a=7^5, c=0, m=2^32-1, output=c(seed)) {
    if (length(output) == n)  # Algoritmo pabaiga (tikrinama terminavimo slyga)
        return(output / m)  # Mainamas reikmi intervalas i [0;m)  [0;1)
    
    new_value = (a * seed + c) %% m  # Tiesinis kongruentinis algoritmas
    
    return(Recall(seed=new_value, output=append(output, new_value), n=n, a=a, c=c, m=m))
}

2.1. Skaičiaus pi vertinimas statistiniu būdu¶

Turime kvadrat su virnmis $(-1,-1)$ ir $(1, 1)$ ir jame brt apskritim. iame plote tolygiu pasiskirstymu atsitiktinse vietose padsime takus. Laikysime, kad tak skaiiaus patenkanio plot santykis su visu tak skaiiumi aproksimuoja to ploto santyk su visu kvadratu. Plot, kuriuos riboja apskritimas ir kvadratas, santykis lygus $\frac{}{4}$, todl iekodami pi reikms apskaiiuot santyk dauginsime i $4$.

Pi įverčio radimo funkcijos implementacija¶

estimate_pi = Vectorize(function(points.count) {
    # Kvadrate padt tak x ir y komponeni generavimas
    points.x = runif(points.count, -1, 1) 
    points.y = runif(points.count, -1, 1)
    
    # Apskaiiuojamas atenkani  apskritim tak kiekio santykis
    points.inside.count = sum(points.x ^ 2 + points.y ^ 2 <= 1)
    points.inside.ratio = points.inside.count / points.count
    
    # Pi verio gavimas
    pi.estimate = points.inside.ratio * 4
    return(pi.estimate)
})

Užrašoma k-osios pozicinės statistikos pas. f-jos išraiška¶

I uduoties apraymo inome: $$F_{X_{(k)}}(x)=\textbf{P}\{X_{(k)}\leqslant x\}=\textbf{P}\{Y \geqslant k\} = \sum\limits_{j=k}^n \binom{n}{j} F(x)^j \big(1-F(x)\big)^{n-j}$$

$F_{X_{(k)}}(x)$ implementuojamas su cdf_k()

In [28]:
cdf_k = function(arg.x, cdf, n, k) {
    # Kadangi pasiskirstymo funkcija yra suma, galime apibrti funkcij rasti jos dmeniui (angl. summand)
    summand = Vectorize(function(j) choose(n, j)  *  cdf(arg.x)^j  *  (1 - cdf(arg.x))^(n - j))
    return(sum(summand(k:n)))
}
x.k.cdf = function(arg.x) cdf_k(arg.x = arg.x, cdf = X.cdf, n = x.size, k = k)

winsorize_sample = function(lst, k) {
    n = length(lst)
    lst = sort(lst)
    lst[1:k] = lst[k+1]
    lst[(n-k+1):n] = lst[n-k]
    return(lst)
}

run_experiment = function(xx, transform_samples_function) {
    xx.transformed = transform_samples_function(xx)
    xx.transformed.mean = Vectorize(function(arg.x) mean(arg.x))(xx.transformed)
    
    # Briama transformuot vidurki histograma
    par(mar = c(5, 4, 4, 4) + .3)  # Paliekame vietos z aiai
    hist(xx.transformed.mean, main="", ylab="Danis", xlab=expression(bar(x)),
         breaks=seq(plot.const.x.min, plot.const.x.max, length.out=plot.const.x.bars))
    
    # Briamas teorinio empirinio vidurkio tankio f-jos grafikas
    par(new = T)  # Brime  jau esant grafik
    plot.y = x.mean.pdf(seq(plot.const.x.min, plot.const.x.max, length.out = 100))
    plot(plot.y, type='l', axes = FALSE, bty = "n", xlab = "", ylab = "", col = "red")
    axis(side = 4, at = pretty(range(plot.y)))
    mtext("Teorinis tankis", side = 4, line = 3)
    legend("topleft", lty = c(1), col="red", legend = c("Teor. emp. vidurk. tank."))
    
    # Apskaiiuojamas transformuoto vidurkio: vidurkis ir dispersija
    return(data.frame(
        transformed_mean_mean=mean(xx.transformed.mean),
        transformed_mean_variance=var(xx.transformed.mean)
    ))
}

estimate_kernel_density = function(z, x, h, kernel) {
    summand = Vectorize(function(x.i)  kernel((z - x.i) / h)  )
    return(sum(summand(x)) / (length(x) * h))
}
                        
# estimate_kernel_density f-jai pritaikomas vektorizuotas kintamj iklimas (angl. currying)
ekd.curried = function(z, kernel) 
                Vectorize(function(h) 
                Vectorize(function(x)
                estimate_kernel_density(z, x, h, kernel)))                        
                        
kernel.uniform = function(arg.x) (abs(arg.x) <= 1) / 2  # Tolygusis branduolys
kernel.gaussian = function(arg.x) exp(-arg.x ^ 2 / 2) / sqrt(2 * pi)  # Gauso branduolys

Patikrinama, ar funkcija estimate_kernel_density() implementuota tinkamai, lyginant j su density() f-ja i standartins bibliotekos.

a) Metodas 1: Parametro $\theta$ įvertis gaunamas per pirmosios eilės pradinį momentą¶

Pasiskirtymo parametras $\theta$ susiejamas su paprastja atsitiktine imtimi $X$ per pirmosios eils pradin moment $\textbf{E}\xi $ ir ireikiama vertinio gavimo formul.

$$\textbf{E}\xi = \frac{a+b}{2} = \frac{0+\theta}{2} \implies \theta = 2\, \textbf{E} \xi \implies \theta' = 2 \overline{X}$$

Pasiskirtymo parametras $\theta$ susiejamas su paprastja atsitiktine imtimi $X$ per antrosios eils pradin moment $\textbf{E}\xi^2 $ ir ireikiama vertinio gavimo formul.

Analiziniu būdu gaunamas $\hat{\lambda}$ įvertis¶

Iekomas logtiktinumo funkcijos ekstremumas (maksimumas), pagal $\lambda$. iame maksimume, $\lambda$ reikm yra iekomasis vertinys. Maksimumui rasti, iekosime sprendinio lygiai $\frac{\partial l(\lambda, x)}{\partial \lambda} = 0$.

$$\frac{\partial l(\lambda, x)}{\partial \lambda} = \frac{\partial}{\partial \lambda} \sum\limits_{i=1}^n \ln p(x_i,\lambda) = \sum\limits_{i=1}^n \frac{\partial}{\partial \lambda} \ln \frac{\lambda^{x_i} e^{-\lambda}}{x_i!} = \sum\limits_{i=1}^n \bigg( \frac{x_i}{\lambda} - 1 \bigg) = \frac{1}{\lambda} \sum\limits_{i=1}^n \big(x_i\big) - n = 0 $$$$\lambda = \frac{1}{n} \sum\limits_{i=1}^n x_i = \textbf{E}X$$

estimate_dist_parameter = function(x, loglikfun.param, use_modified_newton_method=F, epsilon=1e-3, parameter.initial=1) {
    iterations = 0
    parameter = parameter.initial
    
    repeat {
        iterations = iterations + 1
        derivative.first = fderiv(f = loglikfun.param, x = parameter, n = 1)
        derivative.second.x = ifelse(use_modified_newton_method, parameter.initial, parameter)
        derivative.second = fderiv(f = loglikfun.param, x = derivative.second.x, n = 2)
        parameter.new = parameter - derivative.first / derivative.second
        
        if (abs(parameter - parameter.new) < epsilon) {
            parameter = parameter.new
            break
        }
        parameter = parameter.new
    }
    return(data.frame(parameter=parameter,
                      iterations=iterations,
                      epsilon=epsilon,
                      use_modified_newton_method=use_modified_newton_method))
}
